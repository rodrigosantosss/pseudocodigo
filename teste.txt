Algoritmo lojas
Variáveis
    i, loja_max, ganho_max, ganho, soma: inteiro
    localidade: cadeia
Início
    i <- 1
    ganho_max <- 0
    soma <- 0
    Enquanto i < 6 faça
        Escrever('Digite a localidade da loja: ')
        Ler(localidade)
        Se localidade = 'algarve' Então
            Escrever('Digite os ganhos da loja nos últimos seis meses: ')
            Ler(ganho)
            Se ganho > ganho_max Então
                ganho_max <- ganho
                loja_max <- i
            Fim
            Escrever('Ganhos da loja nº', i, ': ', ganho, '€')
            soma <- soma + ganho
            i <- i + 1
        Fim
    Fim
    Escrever('A loja que ganhou mais foi a nº', loja_max)
    Escrever('O total de ganhos das cinco lojas foi ', soma, '€')
Fim.
