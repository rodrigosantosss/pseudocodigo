Algoritmo caracteristicas_terreno
Variáveis
    ladoA, ladoB, perimetro, area: real
    quociente, resto: inteiro
Início
    Ler(ladoA, ladoB)
    area <- ladoA * ladoB
    perimetro <- 2 * (ladoA + ladoB)
    Se area < 100 Então
        Escrever("terreno pequeno")
    Senão
        Escrever("terreno bastante grande")
    Fim
    Escrever("O perímetro é ", perimetro, " e a área é ", area, ".")
Fim.
